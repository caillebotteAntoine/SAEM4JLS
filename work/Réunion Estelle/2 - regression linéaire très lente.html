               mu = 3,  #moyenne de phi
               omega2 = 0.1 )  #variance de phi

Phi <- function(sigma2, mu, omega2)
  c(- 1/(2*sigma2), -1/(2*omega2), mu/omega2)

S <- function(phi)
{
  s <- c(sum((Y - get_obs(m, dt, phi = phi))^2 ),
    apply(phi^2, 2, sum),
    apply(phi  , 2, sum) )

  attr(s, '1') <- 1
  attr(s, '2') <- 1 + 1:ncol(phi)
  attr(s, '3') <- 1 + ncol(phi) + 1:ncol(phi)
  return(s)
}
#=============================================================================#
transi.phi <- function(x) rnorm(length(x), x, 0.2)
loglik.phi <- function(phi, Phi)
{
  s <- S(phi)
  sum(s*Phi)
}

phi.MH <- MH_High_Dim(500, matrix(rnorm(F.*G, 0,1), ncol = F.), transi.phi, loglik.phi, do.call(Phi, param), verbatim = T)
plot(phi.MH, dim = G)

M <- 2
u <- function(k) ifelse(k<100, 1, 1/(k-99) )
#---
para <- list( sigma2 = sample(1:5, size = 1),
              mu = rep(0, F.),
              omega2 = rep(1, F.) )

phigivenY <- 1:M %>% lapply(function(i) matrix(rnorm(F.*G, 0,1), ncol = F.) )
#---
sim <- function(niter, Phih, phigivenY)
{
  M <- length(phigivenY)

  phigivenY <- 1:M %>% lapply( function(i)
    MH_High_Dim(niter, phigivenY[[i]], transi.phi, loglik.phi, Phih ))

  list(phi = phigivenY)
}

maxi <- function(S)
{
  mu <- S[attr(S,'3')]/G # = S[4]
  return(list(sigma2 = S[attr(S,'1')]/n,
              mu = mu,
              omega2 = S[attr(S,'2')]/G - mu^2)
  )
}
