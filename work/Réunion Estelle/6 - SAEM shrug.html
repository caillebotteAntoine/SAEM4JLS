SAEM avec simulation par MCMC

\[\log f(Y, Z ; \theta) = \langle \Phi(\theta) ; S(Y,Z) \rangle - \psi(\theta)\]

Phi <- function(rho2, sigma2, mu, omega2)
  c(- 1/(2*sigma2), -1/(2*rho2), -1/(2*omega2), mu/omega2)

S <- function(eta, phi)
{
  s <- c(sum((Y - get_obs(m, dt, eta = eta, phi = phi) )^2 ), #id = 1
    sum(eta^2),          #id = 2
    apply(phi^2, 2, sum),          #id = 3
    apply(phi  , 2, sum) )         #id = 4

  attr(s, '1') <- 1
  attr(s, '2') <- 2
  attr(s, '3') <- 2 + 1:ncol(phi)
  attr(s, '4') <- 2 +   ncol(phi) + 1:ncol(phi)
  return(s)
}

transi <- function(x) rnorm(length(x), x, 0.05)
loglik.phi <- function(x, eta, Phi)
{
  s <- S(eta,x)
  
  id <- c(attr(s,'1'), attr(s,'3'), attr(s,'4')) #indice de S_1 et S_{3,.} puis S_{4,.}
  sum(Phi[id]*s[id])
}
loglik.eta <- function(x, phi, Phi) 
{ 
  s <- S(x, phi)
  id <- c(attr(s,'1'), attr(s,'2'))
  sum(Phi[id]*s[id])
}

sim <- function(niter, Phih, etagivenY, phigivenY)
{
  M <- length(phigivenY)
  etagivenY <- 1:M %>% lapply( function(i) 
    MH_High_Dim(niter, etagivenY[[i]], transi, loglik.eta, phigivenY[[i]], Phih ))

  phigivenY <- 1:M %>% lapply( function(i)
    MH_High_Dim(niter, phigivenY[[i]], transi, loglik.phi, etagivenY[[i]], Phih ))

  list(eta = etagivenY, phi = phigivenY)
}

maxi <- function(S)
{
  mu <- S[attr(S,'4')]/G # = S[4]
  return( list(rho2 = S[attr(S,'2')]/N,
               sigma2 = S[attr(S,'1')]/n,
               mu = mu,
               omega2 = S[attr(S,'3')]/G - mu^2))
  
}
